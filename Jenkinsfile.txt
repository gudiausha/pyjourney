pipeline {
         agent any
         stages {
                 stage('One') {
                 steps {
                     echo 'Hi, this is Pratyusha'
                 }
                 }
                 stage('Two') {
                 steps {
                    input('Do you want to proceed?')
                 }
                 }
                 stage('Three') {
                 when {
                       not {
                            branch "master"
                       }
                 }
                 steps {
                       echo "When the not condition succeeds this echo gets executed"
                 }
                 }
                 stage('Four') {
                 parallel { 
                            stage('Unit Test') {
                           steps {
                                echo "Running the unit test..."
                           }
                           }
                           }
                           }
               stage('Five'){
               steps {
                retry(2) {
                    echo "Retrying for 2 times"
                }

                timeout(time: 3, unit: 'MINUTES') {
                    echo "Timeout"
                }
                }
                           
              }
           }
}
